3   import java.io.IOException;
4   
5   import okhttp3.Credentials;
6   import okhttp3.Interceptor;
7   import okhttp3.Request;
8   import okhttp3.Response;
9   
10  /*
11   * https://stackoverflow.com/questions/22490057/android-okhttp-with-basic-
12   * authentication
13   */
14  @SuppressWarnings("javadoc")
15  public class BasicAuthInterceptor implements Interceptor {
16  	private final String credentials;
17  
18  	public BasicAuthInterceptor(final String user, final String password) {
19  		this.credentials = Credentials.basic(user, password);
20  	}
21  
22  	@Override
23  	public Response intercept(final Chain chain) throws IOException {
24  		final Request request = chain.request();
25  		final Request authenticatedRequest = request.newBuilder().header("Authorization", credentials).build();
26  		return chain.proceed(authenticatedRequest);
27  	}
28  }
